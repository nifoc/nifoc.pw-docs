Katja VM Stats provides an easy way to send information about the Erlang VM to Riemann. The source code can be found on GitHub.

The public API of Katja VM Stats can be found in the katja_vmstats module.

The default configuration looks like this:

[
  {katja_vmstats, [
    {service, "katja_vmstats"},
    {delay_collection, 0},
    {collector, [
      [
        {interval, 1000},
        {metrics, [
          error_logger_message_queue,
          loaded_modules,
          memory_atoms,
          memory_binaries,
          memory_ets,
          memory_processes,
          memory_system,
          memory_total,
          port_count,
          port_limit,
          port_utilization,
          process_count,
          process_limit,
          process_utilization,
          run_queue
        ]}
      ]
    ]}
  ]}
].
